## 组件代码示例

### PermissionGate 实现

```typescript
// components/PermissionGate.tsx
interface PermissionGateProps {
  permission: string;
  fallback?: React.ReactNode;
  children: React.ReactNode;
}

export const PermissionGate: React.FC<PermissionGateProps> = ({
  permission,
  fallback = null,
  children
}) => {
  const { currentRole } = useAppStore();
  const rbacMatrix = useRBACMatrix();

  const hasPermission = rbacMatrix[currentRole]?.[permission];

  return hasPermission ? <>{children}</> : fallback;
};
```

### RoleBasedView 实现

```typescript
// components/RoleBasedView.tsx
interface RoleBasedViewProps {
  roles: string[];
  children: React.ReactNode;
}

export const RoleBasedView: React.FC<RoleBasedViewProps> = ({
  roles,
  children
}) => {
  const { currentRole } = useAppStore();

  if (!roles.includes(currentRole)) {
    return (
      <Result
        status="403"
        title="无权限访问"
        subTitle={`此视图仅限 ${roles.join('、')} 访问`}
      />
    );
  }

  return <>{children}</>;
};
```

### ActionCatalog 右键菜单实现

```typescript
// components/ActionCatalog.tsx
export const ActionCatalog: React.FC<ActionCatalogProps> = ({
  targetObject,
  permissions
}) => {
  const getAvailableActions = () => {
    const actions = [];
    
    if (permissions.write_S2_CONSTRAINT) {
      actions.push({
        key: 'review',
        label: '发起人工审核',
        icon: <CheckCircle />
      });
      actions.push({
        key: 'arbitrate',
        label: '冲突仲裁',
        icon: <SwapRight />
      });
    }
    
    if (permissions.write_S4_DECISION) {
      actions.push({
        key: 'approve',
        label: '批准决策',
        icon: <CheckOutlined />
      });
    }
    
    return actions;
  };

  return (
    <Dropdown
      menu={{
        items: getAvailableActions()
      }}
      trigger={['contextMenu']}
    >
      {children}
    </Dropdown>
  );
};
```
